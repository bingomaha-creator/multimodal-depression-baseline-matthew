# 4. MLP smoke test: audio + visual, one epoch and two optimizer steps

  --model mlp \
  --modality both \
  --seed 42 \
  --device cuda \
  --epochs 1 \
  --max-train-steps 2 \
  --output-dir runs/D-vlog-smoke

  --modality both \
  --seed 42 \
  --device cuda \
  --epochs 1 \
